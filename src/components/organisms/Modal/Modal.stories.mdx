<Preview>
  <Story name="default">
    {() => {
      const [isShowing, setIsShowing] = useState(false)
      return (
      <div>
       <Button onClick={()=>{setIsShowing(true); action('modal-open') }}>Open Modal</Button>
       <Modal
        toggled={isShowing}  
        title={<span>Confirmation</span>}
        body={<span>"Your community has been successfully registered. A verification link has been sent to your email account. Please verify your email in order to access the created community."</span>}
        footer={<Button type="submit" onClick={()=>{isShowing===true ? setIsShowing(false) : null}}>Ok</Button>}
        onClickOutOfModal={()=>{setIsShowing(false)}}
        />
    </div>)
    }}
  </Story>
